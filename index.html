            beta = 0.5 + np.random.uniform(-0.5, 0.5)   # 对比度

            image = cv2.convertScaleAbs(image, alpha=alpha, beta=beta)

            # 调整饱和度
            hsv[:, :, 1] = np.clip(hsv[:, :, 1] * saturation_scale, 0, 255)
            distorted_image = cv2.cvtColor(hsv, cv2.COLOR_HSV2BGR)
